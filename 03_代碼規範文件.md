### 4.1 LLM Provider Interface
```python
from abc import ABC, abstractmethod

class LLMProvider(ABC):
    @abstractmethod
    async def generate_summary(self, content: str) -> dict:
        """
        輸入：原始文字內容
        輸出：{"summary": str, "tags": list, "category": str}
        """
        pass

    @abstractmethod
    async def chat(self, messages: list) -> str:
        """
        輸入：對話歷史 [{"role": "user", "content": "..."}]
        輸出：AI 回覆文字
        """
        pass
```

### 4.2 Factory Pattern (AI Service) - 10 Models Support
```python
from .llm.gemini import GeminiProvider
from .llm.openai import OpenAIProvider
from .llm.ollama import OllamaProvider
from .llm.claude import ClaudeProvider
from .llm.deepseek import DeepSeekProvider
from .llm.mistral import MistralProvider
from .llm.groq import GroqProvider
from .llm.openrouter import OpenRouterProvider
from .llm.kimi import KimiProvider
from .llm.qwen import QwenProvider
from config import settings

def get_llm_provider() -> LLMProvider:
    provider_type = settings.ai_provider.lower()
    
    providers = {
        "gemini": lambda: GeminiProvider(api_key=settings.google_api_key),
        "openai": lambda: OpenAIProvider(api_key=settings.openai_api_key),
        "ollama": lambda: OllamaProvider(base_url=settings.ollama_base_url),
        "claude": lambda: ClaudeProvider(api_key=settings.anthropic_api_key),
        "deepseek": lambda: DeepSeekProvider(api_key=settings.deepseek_api_key),
        "mistral": lambda: MistralProvider(api_key=settings.mistral_api_key),
        "groq": lambda: GroqProvider(api_key=settings.groq_api_key),
        "openrouter": lambda: OpenRouterProvider(api_key=settings.openrouter_api_key),
        "kimi": lambda: KimiProvider(api_key=settings.moonshot_api_key),
        "qwen": lambda: QwenProvider(api_key=settings.dashscope_api_key),
    }
    
    if provider_type in providers:
        return providers[provider_type]()
    else:
        raise ValueError(f"Unsupported AI Provider: {provider_type}")
```

### 4.3 Salesforce Slack 訊息解析
```python
import re

def parse_salesforce_message(text: str) -> dict:
    """
    解析從 Slack 轉發過來的 Salesforce 訊息
    格式範例：
    Opportunity: TSMC Expansion
    Amount: $5,000,000
    Stage: Negotiation
    """
    data = {}
    
    if match := re.search(r"Opportunity:\s*(.+)", text):
        data["title"] = match.group(1).strip()
    if match := re.search(r"Amount:\s*[\$]?([\d,]+)", text):
        data["amount"] = match.group(1).replace(",", "")
    if match := re.search(r"Stage:\s*(.+)", text):
        data["stage"] = match.group(1).strip()
        
    return data
```
